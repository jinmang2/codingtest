```python
from typing import List


num2hand = dict(enumerate(" L RL RL R"))
num2hand.update({"*": "L", "#": "R"})
num2cart = {
    1: [0, 0], 2: [0, 1], 3: [0, 2],
    4: [1, 0], 5: [1, 1], 6: [1, 2],
    7: [2, 0], 8: [2, 1], 9: [2, 2],
    "*": [3, 0], 0: [3, 1], "#": [3, 2],
}


def distance(x: List[int], y: List[int]):
    return sum([abs(i-j) for i, j in zip(x, y)])


def solution(numbers: List[int], hand: str) -> str:
    answer = []
    left, right = "*", "#"
    for num in numbers:
        _hand = num2hand[num]
        if _hand == " ":
            distL = distance(num2cart[num], num2cart[left])
            distR = distance(num2cart[num], num2cart[right])
            if distL == distR:
                _hand = hand.title()[0]
                if hand == "right":
                    right = num
                else:
                    left = num
            elif distL < distR:
                _hand = "L"
                left = num
            elif distL > distR:
                _hand = "R"
                right = num
        else:
            if _hand == "R":
                right = num
            else:
                left = num
        answer.append(_hand)
    return "".join(answer)
```
